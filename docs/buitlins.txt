[Built-ins]

# Basc CLI Constants

// List of string arguments.
var argv: "opaque:list";
// Number of given arguments.
var argc: "int";

# Text IO

println(value: "any...") -> "null"
    A variadic function that prints values joined with spaces.
    The last item will be printed with a new line.

print(value: "any...") -> "null"
    A variadic function that prints values joined with spaces.
    The last item does not have a trailing new line.

printr(value: "any...") --> "null"
    A variadic function that prints values not joined with spaces.
    The last item doesnt contain a trailing new line.

input(prompt: "string?") -> "string"
    A function that prompts the user for input.

# File IO

var stdin: "opaque:stdin fd";
var stdout: "opaque:stdout fd";
var SEEK_END: "int";
var SEEK_SET: "int";

open(file: "string", mode: "string") -> "opaque:file"
    A function that opens a file, a thin wrapper around 'fopen'.

fread(file: "opaque:file", size: "int") -> "string"
    A function that reads a specified number of bytes 'size'.
    A thin wrapper around 'fread'.

fprint(file: "opaque:file", value: "string...") -> "null"
    A function that prints the soecified values into the file.

fseek(file: "opaque:file", whence: "int", pos: "int") -> "null"
    See C function 'fseek'.

ftell(file: "opaque:file") -> "int"
    See C function 'ftell'.

fclose(file: "opaque:file") -> "null"
    A function that closes the opened file handle.

# Platform Specific

sys.get_arch() -> "string"
    Get the architecture of the platform MiLa was built for

sys.get_platform() -> "string"
    Get the platform MiLa is compiled for (if any)

system(command: "string") -> "int"
    Run the comand a if typed in a terminal.
    This function may not be available if the system
    does not expose this feature (I.E. in embedded contexts)
    The function returns the return code of the command ran.

# List operations

list(items: "any...") -> "opaque:list"
    Creates a list of the given items.

list.append(list: "opaque:list", item: "any") -> "null"
    Append the item to the list.

list.insert(list: "opaque:list", index: "int", item: "any") -> "null"
    Insert the item to the specific index, if the index dosnt exist
    it will not append nor prepend the item but produces an error.

list.rem(list: "opaque:list", index: "int") -> "null"
    Remove the given index from the list.

list.pop(list: "opaque:list") -> "any"
    Pop the last item and return it.
